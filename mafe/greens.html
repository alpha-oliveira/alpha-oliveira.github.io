def plot_green(H, init=-15, fin= -5, steps = 500, eta = 1e-10j, title = '|G(E)|' ):
    d, V = eig(H)
    epsilons = linspace(init, fin, steps)
    epsilons = epsilons + eta*ones(epsilons.shape)
    ketbras = array( [v.reshape(-1, 1) @ v.reshape(1, -1) for v in V.T])
    Gs = [ketbras.T.dot(1/(epsilon - d)) for epsilon in epsilons]
    G1N = [sqrt(g[0,-1]*conj(g[0,-1])) for g in Gs]
    del Gs
    fig, ax = subplots()
    ax.plot(epsilons,G1N)
    ax.set_title(title)
    return G1N

def single_plot(H, init=-15, fin= -5, steps = 500, eta = 1e-10j, title = '|G(E)|', label='molecule' ):
    d, V = eig(H)
    epsilons = linspace(init, fin, steps)
    epsilons = epsilons + eta*ones(epsilons.shape)
    ketbras = array( [v.reshape(-1, 1) @ v.reshape(1, -1) for v in V.T])
    Gs = [ketbras.T.dot(1/(epsilon - d)) for epsilon in epsilons]
    G1N = [sqrt(g[0,-1]*conj(g[0,-1])) for g in Gs]
    del Gs
    plot(epsilons,G1N, label=label)
    legend()
    return G1N
